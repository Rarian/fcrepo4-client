16  package org.fcrepo.client;
17  
18  import java.util.Collection;
19  
20  /**
21   * A Fedora Object, a Resource which can contain Datastreams and/or other Objects.
22   * @author escowles
23   * @since 2014-08-01
24  **/
25  public interface FedoraObject extends FedoraResource {
26  
27      /**
28       * Get the Object and Datastream nodes that are children of the current Object.
29       * @param mixin If not null, limit to results that have this mixin.
30      **/
31      public Collection<FedoraResource> getChildren( String mixin ) throws FedoraException;
32  
33      /**
34       * Create a new resource with a repository-supplied path contained within
35       * the resource exposed by this FedoraObject instance.
36       * @return a FedoraObject representing the created resource.
37       * @throws FedoraException if an error occurs while making the requests against the repository.
38       **/
39      public FedoraObject createObject() throws FedoraException;
40  }
