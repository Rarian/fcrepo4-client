016package org.fcrepo.client;
017
018import java.util.Collection;
019
020/**
021 * A Fedora Object, a Resource which can contain Datastreams and/or other Objects.
022 * @author escowles
023 * @since 2014-08-01
024**/
025public interface FedoraObject extends FedoraResource {
026
027    /**
028     * Get the Object and Datastream nodes that are children of the current Object.
029     * @param mixin If not null, limit to results that have this mixin.
030    **/
031    public Collection<FedoraResource> getChildren( String mixin ) throws FedoraException;
032
033    /**
034     * Create a new resource with a repository-supplied path contained within
035     * the resource exposed by this FedoraObject instance.
036     * @return a FedoraObject representing the created resource.
037     * @throws FedoraException if an error occurs while making the requests against the repository.
038     **/
039    public FedoraObject createObject() throws FedoraException;
040}
