016package org.fcrepo.client.impl;
017
018import static org.fcrepo.kernel.RdfLexicon.CONTAINS;
019import static org.fcrepo.kernel.RdfLexicon.HAS_MIXIN_TYPE;
020
021import java.util.Collection;
022import java.util.HashSet;
023import java.util.Set;
024
025import com.hp.hpl.jena.graph.Node;
026import com.hp.hpl.jena.graph.NodeFactory;
027import com.hp.hpl.jena.graph.Triple;
028import com.hp.hpl.jena.util.iterator.ExtendedIterator;
029
030import org.fcrepo.client.FedoraException;
031import org.fcrepo.client.FedoraObject;
032import org.fcrepo.client.FedoraRepository;
033import org.fcrepo.client.FedoraResource;
034import org.fcrepo.client.utils.HttpHelper;
035
036/**
037 * A Fedora Object Impl.
038 *
039 * @author lsitu
040 * @author escowles
041 * @since 2014-08-11
042 */
043public class FedoraObjectImpl extends FedoraResourceImpl implements FedoraObject {
044    private final static Node binaryType = NodeFactory.createLiteral("fedora:binary");
045
046    /**
047     * Constructor for FedoraObjectImpl
048     *
049     * @param repository FedoraRepository that created this object
050     * @param httpHelper HTTP helper for making repository requests
051     * @param path Repository path
052     */
053    public FedoraObjectImpl(final FedoraRepository repository, final HttpHelper httpHelper, final String path) {
054        super(repository, httpHelper, path);
055    }
056
057    /**
058     * Get the Object and Datastream nodes that are children of the current Object.
059     *
060     * @param mixin If not null, limit to results that have this mixin.
061     */
062    public Collection<FedoraResource> getChildren(final String mixin) throws FedoraException {
063        Node mixinLiteral = null;
064        if ( mixin != null ) {
065            mixinLiteral = NodeFactory.createLiteral(mixin);
066        }
067        final ExtendedIterator<Triple> it = graph.find(Node.ANY, CONTAINS.asNode(), Node.ANY);
068        final Set<FedoraResource> set = new HashSet<>();
069        while (it.hasNext()) {
070            final Node child = it.next().getObject();
071            if ( mixin == null || graph.contains(child, HAS_MIXIN_TYPE.asNode(), mixinLiteral) ) {
072                final String path = child.getURI().toString()
073                        .replaceAll(repository.getRepositoryUrl(),"");
074                if ( graph.contains(child, HAS_MIXIN_TYPE.asNode(), binaryType) ) {
075                    set.add( repository.getDatastream(path) );
076                } else {
077                    set.add( repository.getObject(path) );
078                }
079            }
080        }
081        return set;
082    }
083}
