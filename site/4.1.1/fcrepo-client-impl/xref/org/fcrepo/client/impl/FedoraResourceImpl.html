168     @Override
169     public void updateProperties(final String sparqlUpdate) throws FedoraException {
170         final HttpPatch patch = httpHelper.createPatchMethod(path, sparqlUpdate);
171 
172         try {
173             final HttpResponse response = httpHelper.execute( patch );
174             final StatusLine status = response.getStatusLine();
175             final String uri = patch.getURI().toString();
176 
177             if ( status.getStatusCode() == SC_NO_CONTENT) {
178                 LOGGER.debug("triples updated successfully for resource {}", uri);
179             } else if ( status.getStatusCode() == SC_FORBIDDEN) {
180                 LOGGER.error("updating resource {} is not authorized.", uri);
181                 throw new ForbiddenException("updating resource " + uri + " is not authorized.");
182             } else if ( status.getStatusCode() == SC_NOT_FOUND) {
183                 LOGGER.error("resource {} does not exist, cannot update", uri);
184                 throw new NotFoundException("resource " + uri + " does not exist, cannot update");
185             } else if ( status.getStatusCode() == SC_CONFLICT) {
186                 LOGGER.error("resource {} is locked", uri);
187                 throw new FedoraException("resource is locked: " + uri);
188             } else {
189                 LOGGER.error("error updating resource {}: {} {}", uri, status.getStatusCode(),
190                              status.getReasonPhrase());
191                 throw new FedoraException("error updating resource " + uri + ": " + status.getStatusCode() + " " +
192                                           status.getReasonPhrase());
193             }
194 
195             // update properties from server
196             httpHelper.loadProperties(this);
197 
198         } catch (final FedoraException e) {
199             throw e;
200         } catch (final Exception e) {
201             LOGGER.error("could not encode URI parameter", e);
202             throw new FedoraException(e);
203         } finally {
204             patch.releaseConnection();
205         }
206     }

250     @Override
251     public boolean isWritable() {
252         final Collection<String> values = getPropertyValues(RdfLexicon.WRITABLE);
253         if (values != null && values.size() > 0) {
254             final Iterator<String> it = values.iterator();
255             return Boolean.parseBoolean(it.next());
256         }
257         return false;
258     }

260     @Override
261     public void createVersionSnapshot(final String label) throws FedoraException {
262         final HttpPost postVersion = httpHelper.createPostMethod(path + "/fcr:versions", null);
263         try {
264             postVersion.setHeader("Slug", label);
265             final HttpResponse response = httpHelper.execute(postVersion);
266             final StatusLine status = response.getStatusLine();
267             final String uri = postVersion.getURI().toString();
268 
269             if ( status.getStatusCode() == SC_NO_CONTENT) {
270                 LOGGER.debug("new version created for resource at {}", uri);
271             } else if ( status.getStatusCode() == SC_CONFLICT) {
272                 LOGGER.debug("The label {} is in use by another version.", label);
273                 throw new FedoraException("The label \"" + label + "\" is in use by another version.");
274             } else if ( status.getStatusCode() == SC_FORBIDDEN) {
275                 LOGGER.error("updating resource {} is not authorized.", uri);
276                 throw new ForbiddenException("updating resource " + uri + " is not authorized.");
277             } else if ( status.getStatusCode() == SC_NOT_FOUND) {
278                 LOGGER.error("resource {} does not exist, cannot create version", uri);
279                 throw new NotFoundException("resource " + uri + " does not exist, cannot create version");
280             } else {
281                 LOGGER.error("error updating resource {}: {} {}", uri, status.getStatusCode(),
282                         status.getReasonPhrase());
283                 throw new FedoraException("error updating resource " + uri + ": " + status.getStatusCode() + " " +
284                         status.getReasonPhrase());
285             }
286         } catch (IOException e) {
287             LOGGER.error("Error executing request", e);
288             throw new FedoraException(e);
289         } finally {
290             postVersion.releaseConnection();
291         }
292     }

310     private Date getDate(final Property property) {
311         Date date = null;
312         final Triple t = getTriple(subject, property);
313         if ( t != null ) {
314             final String dateValue = t.getObject().getLiteralValue().toString();
315             try {
316                 date = dateFormat.parse(dateValue);
317             } catch (final ParseException e) {
318                 LOGGER.debug("Invalid date format error: " + dateValue);
319             }
320         }
321         return date;
322     }

324     /**
325      * Return all the values of a property
326      *
327      * @param property The Property to get values for
328      * @return Collection of values
329      */
330     protected Collection<String> getPropertyValues(final Property property) {
331         final ExtendedIterator<Triple> iterator = graph.find(Node.ANY,
332                                                              property.asNode(),
333                                                              Node.ANY);
334         final Set<String> set = new HashSet<>();
335         while (iterator.hasNext()) {
336             final Node object = iterator.next().getObject();
337             if (object.isLiteral()) {
338                 set.add(object.getLiteralValue().toString());
339             } else if (object.isURI()) {
340                 set.add(object.getURI().toString());
341             } else {
342                 set.add(object.toString());
343             }
344         }
345         return set;
346     }
347 
348     protected Triple getTriple( final Node subject, final Property property ) {
349         final ExtendedIterator<Triple> it = graph.find( subject, property.asNode(), null );
350         try {
351             if ( it.hasNext() ) {
352                 return it.next();
353             } else {
354                 return null;
355             }
356         } finally {
357             it.close();
358         }
359     }
